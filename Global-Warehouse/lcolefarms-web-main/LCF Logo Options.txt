    <script>
        function toggleLogo(id) {
            const el = document.getElementById('logo-' + id);
            const btn = document.getElementById('btn-' + id);
            
            if (el.classList.contains('hidden-logo')) {
                el.classList.remove('hidden-logo');
                btn.classList.remove('inactive');
                btn.classList.add('active');
            } else {
                el.classList.add('hidden-logo');
                btn.classList.remove('active');
                btn.classList.add('inactive');
            }
        }

        function showAll() {
            const ids = ['text', 'leaf', 'grain', 'tractor', 'cow', 'hog'];
            ids.forEach(id => {
                document.getElementById('logo-' + id).classList.remove('hidden-logo');
                const btn = document.getElementById('btn-' + id);
                btn.classList.remove('inactive');
                btn.classList.add('active');
            });
        }

        function hideAll() {
            const ids = ['text', 'leaf', 'grain', 'tractor', 'cow', 'hog'];
            ids.forEach(id => {
                document.getElementById('logo-' + id).classList.add('hidden-logo');
                const btn = document.getElementById('btn-' + id);
                btn.classList.remove('active');
                btn.classList.add('inactive');
            });
        }
    </script>
